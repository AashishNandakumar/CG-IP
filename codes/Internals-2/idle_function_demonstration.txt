#include <GL/glut.h>

float angle = 0.0f;

void display(){
    glClear(GL_COLOR_BUFFER_BIT);
    glLoadIdentity();

    glRotatef(angle, 0, 0, 1);

    glBegin(GL_TRIANGLES);
    glVertex2f(-0.5, -0.5);
    glVertex2f(0.5, -0.5);
    glVertex2f(0.0, 0.5);
    glEnd();

    glutSwapBuffers();
}

void idle(){
    angle += 0.1;
    if (angle > 360.0)
        angle -= 360.0;
    glutPostRedisplay();
}

int main(int argc, char **argv){
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);

    glutInitWindowSize(800, 600);
    glutCreateWindow("GLUT IDLE fxn demonstration");

    glutDisplayFunc(display);
    glutIdleFunc(idle);

    glutMainLoop();
    return 0;
}
