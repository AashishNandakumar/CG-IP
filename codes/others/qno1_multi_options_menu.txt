#include <GL/glut.h>
#include <cstdlib>

int width = 900, height = 450, X = 0, Y = 0, factor = 0;
float red = 1, green = 1, blue = 1;

void init(){
    glClearColor(0,0,0,1);

    gluOrtho2D(-800, 800, -800, 800);
}

void display(){
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f(red, green, blue);

    glBegin(GL_POLYGON);
        glVertex2i(0-factor + X, 0-factor + Y);
        glVertex2i(100+factor + X, 0-factor + Y);
        glVertex2i(100+factor + X, 100+factor + Y);
        glVertex2i(0-factor + X, 100+factor + Y);
    glEnd();
    glFlush();
}

void handleKeyboard(unsigned char key, int x, int y) {
    if (key == 'w')
        Y += 10;
    else if (key == 's')
        Y -= 10;
    else if (key == 'a')
        X -= 10;
    else if (key == 'd')
        X += 10;
    else if (key == 'I')
        factor += 10;
    else if (key == 'D')
        factor -= 10;
    else if (key == 'R')
        factor = 0;
    else if (key == 'r') {
        red = 1;
        green = blue = 0;
    } else if (key == 'b') {
        blue = 1;
        green = red = 0;
    } else if (key == 'g') {
        green = 1;
        red = blue = 0;
    } else if (key == 'q')
        exit(0);

    glutPostRedisplay();
}


int main(int argc, char ** argv){

    glutInit(&argc, argv);

    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);

    glutInitWindowPosition(410, 310);
    glutInitWindowSize(width, height);

    glutCreateWindow("Menu");

    init();

    glutDisplayFunc(display);
    glutKeyboardFunc(handleKeyboard);

    glutMainLoop();

    return 0;
}
